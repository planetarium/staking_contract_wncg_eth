/**
 * @title Launchpad
 * @author Planetarium
 * @notice launchpad contract
 */
contract Launchpad is Initializable {
  using SafeMath for uint128;
  using SafeMath for uint256;
  using SafeERC20Upgradeable for IERC20Upgradeable;
 
  uint128 PROJECT_ID;
 
  /// @notice Operator who can manage configuration of this contract.
  address public OPERATOR;
 
  /// @notice TODO
  address public AUCTION_TOKEN_VAULT;
 
  /* ========== EVENTS ========== */
  event OperatorChanged(address indexed _to);
  event LaunchpadWalletChanged(address indexed _address);
  event AuctionTokenVaultChanged(address indexed _address);
 
  /* ========== INITIALIZE ========== */
  function initialize(
    address _operator
  ) external initializer {
    OPERATOR = _operator;
  }
 
  modifier onlyOperator {
    require(msg.sender == OPERATOR, 'ONLY_OPERATOR');
    _;
  }
 
  /**
   * @dev Change Operator
   * @param _to new opertor address
   */
  function changeOperator(address _to) external onlyOperator {
    OPERATOR = _to;
    emit OperatorChanged(_to);
  }
 
  /**
   * @dev Change Auction Token Vault
   * @param _address new auction token vault address
   */
  function changeAuctionTokenVault(address _address) external onlyOperator {
    AUCTION_TOKEN_VAULT = _address;
    emit AuctionTokenVaultChanged(_address);
  }
 
  struct ProjectData {
    bool initialized;
    BatchAuction batchAuction; 
  }
 
  /// @notice TODO
  mapping(uint128 => ProjectData) public PROJECTS;
 
  /// @dev vesting period in second
  uint128 public VESTING_PERIOD = 60 days;
 
  /// @dev token claim period in second
  uint128 public CLIAMABLE_PERIOD = 90 days;
 
  /**
   * @dev TODO: comment update
   */
  function initProject(
    uint128 _projectId,
    address _commitCurrencyAddress,
    IERC20 _auctionToken,
    uint128 _startTime,
    uint128 _endTime,
    uint256 _totalOfferingTokens,
    uint256 _minimumCommitmentAmount,
    address _treasury,
    address payable _auctionWallet
  ) external onlyOperator {
    require(!PROJECTS[_projectId].initialized, "PROJECT_ID_ALREADY_TAKEN");
    
    BatchAuction _auction = new BatchAuction();
    _auction.initAuction(
      _commitCurrencyAddress,
      _auctionToken,
      AUCTION_TOKEN_VAULT,
      _startTime,
      _endTime,
      _totalOfferingTokens,
      _minimumCommitmentAmount,
      _treasury,
      _auctionWallet
    );
 
    PROJECTS[_projectId].batchAuction = _auction;
 
  }

   */

  }

   * @dev Withdraw ETH early (during the offering period)
   * @param _amount Amount that the user tries to withdraw early
   *
   * Requirements:
   *  - the offering must be ongoing
   *  - Amount to withdraw must be less than withdrawable amount
   */
  function withdraw(uint256 _amount) external {
  }
